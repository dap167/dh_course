xquery version "3.0";
declare namespace tei="http://www.tei-c.org/ns/1.0";
declare default element namespace "http://www.w3.org/1999/xhtml";
declare variable $collection := collection('/db/course/folk');
declare variable $songs := $collection/songs;
declare variable $lang := data($songs);
<html>
    <head><title>Songs</title></head>
    <body>{
for $lang in $songs
let $langName := $lang/@language/string()
let $title := $lang/song/meta/title/string()
let $artist := $lang/song/meta/artist/string()
return (<h1>{$langName}</h1>,
        <ol>
            <li>{concat ($title,$artist)}</li>
        </ol>)
    }</body>
</html>
